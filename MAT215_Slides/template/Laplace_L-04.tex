\begin{frame}
    \frametitle{First Translation}
    First Translation Theorem states that:
    \[
        \mathcal L^{-1}\left\{F(s)\right\} = e^{at}\mathcal L^{-1}{F(s+a)}
    \]
    \begin{example}
        \begin{itemize}[label=\textbullet]
            \item Evaluate $$\mathcal L^{-1}\left\{\frac{6}{(s-2)^6}\right\}$$
            \item Evaluate $$\mathcal L^{-1}\left\{\frac{5}{(s-3)^2+15}\right\}$$
            \item Evaluate $$\mathcal L^{-1}\left\{\frac{4s}{(s-3)^2+25}\right\}$$
        \end{itemize}
    \end{example}
\end{frame}

\begin{frame}
    \frametitle{Second Translation}
    Second Translation Theorem states that:
    \[
        \mathcal{L} \left\{f(t-a)u(t-a)\right\} = e^{-as}\mathcal L\left\{f(t+a)\right\}
    \]
    \[
        \mathcal L^{-1}\left\{F(s)e^{-as}\right\} = f(t-a)u(t-a)
    \]
    \begin{example}
        \begin{itemize}[label=\textbullet]
            \item Evaluate $$\mathcal L^{-1}\left\{\frac{1}{s^4}e^{-2s}\right\}$$
            \item Evaluate $$\mathcal L^{-1}\left\{\frac{1}{(s-3)e^{-2s}}\right\}$$
            \item Evaluate $$\mathcal L^{-1}\left\{\frac{1}{s^2+4}e^{-\pi s}\right\}$$
        \end{itemize}
    \end{example}
\end{frame}

\begin{frame}
    \frametitle{Using Partial Fraction Decomposition}
    \begin{example}
        \begin{itemize}[label=\textbullet]
            \item Find the partial fraction decomposition of $$\frac{1}{s(s+1)}$$
            \item Evaluate $$\mathcal L^{-1}\left\{\frac{1}{s(s+1)}e^{-s}\right\}$$
        \end{itemize}
        \textbf{Hint:}
        $$\frac{1}{s(s+1)} = \frac{1}{s} - \frac{1}{s+1}$$
    \end{example}
\end{frame}
\begin{frame}
    \frametitle{Using Partial Fraction Decomposition}
    \begin{example}
        \begin{itemize}[label=\textbullet]
            \item Find the partial fraction decomposition of $$\frac{2s+3}{(s+1)(2s^3+3s^2-3s-2)}$$
            \item Evaluate $$\mathcal L^{-1}\left\{\frac{2s+3}{(s+1)(2s^3+3s^2-3s-2)}\right\}$$
        \end{itemize}
        \textbf{Hint:}
        $$\frac{2s+3}{(s+1)(2s^3+3s^2-3s-2)} = \frac{1/9}{s+1}+\frac{5/18}{s+2}+\frac{1/2}{s-1}+\frac{-16/9}{2s+1}$$
    \end{example}
\end{frame}

            \item Evaluate $$\mathcal L^{-1}\left\{\frac{4s^2-5s}{(s+1)(s-2)^2}\right\}$$

    \end{example}
\end{frame}

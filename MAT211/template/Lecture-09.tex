\begin{frame}[Examples]
    \begin{problem}
        $f(z)=z^2$, find the derivative at $x=3+2i$.  
    \end{problem}
    Using the definition, $$\lim_{\Delta z}\rightarrow 0\frac{f(z+\Delta z)-f(z)}{\Delta z}$$
    make your life easy enough. 
    \begin{problem}
        $f(z)=\bar z$, find the derivative at $x=3+2i$. 
    \end{problem}
    Using the definition, $$\lim_{\Delta z}\rightarrow 0\frac{f(z+\Delta z)-f(z)}{\Delta z}$$
    and use $\Delta z=\Delta x+i\Delta z$ when needed. 
    \begin{problem}
        Using definition show that $f(z)=\frac{2z-3i}{3z-2i}$ is differentiable at $z=-i$.
    \end{problem}
\end{frame}

\question[10 Marks]
\begin{problem}
Let 
\[
f : \mathbb{R} \longrightarrow \mathbb{R}, \qquad f(x) = x^2 - 4x + 3.
\]

\begin{itemize}
\item 
Is $f$ \textbf{onto} (surjective) as a function from $\mathbb{R}$ to $\mathbb{R}$? 
If not, justify your answer and find the \textbf{range} of $f$.

\vspace{1cm}

\item 
Find a restriction of the \textbf{codomain} (say $B \subset \mathbb{R}$) such that 
\[
f : \mathbb{R} \to B
\]
becomes \textbf{onto}. 
Explain why your chosen $B$ works.

\vspace{1cm}

\item 
Now restrict both the \textbf{domain and codomain} so that $f$ becomes a \textbf{bijection}. 
Find such subsets $A \subset \mathbb{R}$ and $B \subset \mathbb{R}$ and justify your answer with reasoning or a sketch.

\vspace{1cm}

\end{itemize}
\end{problem}

\question[10 Points]
\begin{problem}
Let 
\[
A = \{1,2,3\}, \qquad 
R = \{(1,2), (2,3)\}.
\begin{itemize}
\item 
List all ordered pairs that must be \textbf{added} to $R$ to make it \textbf{reflexive} on $A$.
\vspace{1cm}

\item 
List all ordered pairs that must be \textbf{added} to $R$ to make it \textbf{symmetric}.

\vspace{1cm}

\item 
List all ordered pairs that must be \textbf{added} to $R$ to make it \textbf{transitive}.

\vspace{1cm}

\item 
Find the \textbf{smallest relation} on $A$ that is simultaneously 
\textbf{reflexive, symmetric, and transitive} 
(i.e.\ the equivalence closure of $R$).

\vspace{1cm}
\end{itemize}

\end{problem}

\section{Intuition of Continuity}

\begin{frame}{What does continuity mean?}
Informally, a function is \textbf{continuous at a point} if:
\begin{quote}
You can draw its graph near that point without lifting your pen.
\end{quote}

\pause
Intuitively:
\begin{itemize}
  \item No holes
  \item No jumps
  \item No infinite breaks
\end{itemize}
\end{frame}

\begin{frame}{Limit-based definition (core idea)}
A function $f(x)$ is \textbf{continuous at $x=a$} if:
\[
\lim_{x\to a} f(x) = f(a).
\]

\pause
This means three things must hold:
\begin{itemize}[label=(\roman*)] 
  \item $f(a)$ is defined
  \item $\lim_{x\to a} f(x)$ exists
  \item $\lim_{x\to a} f(x) = f(a)$
\end{itemize}
\end{frame}

\section{Basic Examples of Continuity}

\begin{frame}{Example 1: Polynomial functions}
Consider:
\[
f(x) = 3x^2 - 5x + 1.
\]

\pause
\textbf{Solution:}
Polynomials are continuous for all real numbers.

\[
\lim_{x\to a} f(x) = f(a) \quad \text{for all } a\in\mathbb{R}.
\]

\pause
\[
\Rightarrow \text{Continuous everywhere.}
\]
\end{frame}

\begin{frame}{Example 2: Rational function (continuous domain)}
Consider:
\[
f(x)=\frac{x^2-1}{x-1}.
\]

\pause
Simplify:
\[
f(x)=x+1 \quad (x\neq 1).
\]

\pause
At $x=1$:
\[
f(1)\ \text{is undefined}.
\]

\pause
\[
\Rightarrow f \text{ is continuous for } x\neq 1, \text{ but discontinuous at } x=1.
\]
(Removable discontinuity)
\end{frame}

\begin{frame}{Example 3: Absolute value}
Consider:
\[
f(x)=|x|.
\]

\pause
Check continuity at $x=0$:
\[
\lim_{x\to 0^-}|x|=0, \quad \lim_{x\to 0^+}|x|=0.
\]

\pause
Since $f(0)=0$,
\[
\lim_{x\to 0}f(x)=f(0).
\]

\pause
\[
\Rightarrow f(x)=|x| \text{ is continuous everywhere.}
\]
\end{frame}

\section{Discontinuities}

\begin{frame}{Example 4: Jump discontinuity}
Define:
\[
f(x)=
\begin{cases}
2x, & x<1,\\
5, & x\ge 1.
\end{cases}
\]

\pause
Compute:
\[
\lim_{x\to 1^-}f(x)=2,\qquad
\lim_{x\to 1^+}f(x)=5.
\]

\pause
Since left $\neq$ right,
\[
\lim_{x\to 1}f(x) \text{ does not exist}.
\]

\pause
\[
\Rightarrow f \text{ is discontinuous at } x=1.
\]
\end{frame}

\begin{frame}{Example 5: Infinite discontinuity}
Consider:
\[
f(x)=\frac{1}{x}.
\]

\pause
At $x=0$:
\[
\lim_{x\to 0^+}f(x)=+\infty,\quad
\lim_{x\to 0^-}f(x)=-\infty.
\]

\pause
\[
\Rightarrow f \text{ has an infinite discontinuity at } x=0.
\]
\end{frame}

\begin{frame}{Example 6: Oscillatory discontinuity}
Consider:
\[
f(x)=\sin\left(\frac{1}{x}\right).
\]

\pause
As $x\to 0$, the function oscillates infinitely between $-1$ and $1$.

\pause
\[
\lim_{x\to 0} f(x) \text{ does not exist}.
\]

\pause
\[
\Rightarrow f \text{ is discontinuous at } x=0.
\]
\end{frame}

\section{Continuity on Intervals}

\begin{frame}{Continuity on intervals}
A function is:
\begin{itemize}
  \item Continuous on $(a,b)$ if continuous at every point in the interval
  \item Continuous on $[a,b]$ if:
  \begin{itemize}
    \item Continuous on $(a,b)$
    \item Right-continuous at $a$
    \item Left-continuous at $b$
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Example 7: Endpoint continuity}
Consider:
\[
f(x)=\sqrt{x}, \quad x\in[0,4].
\]

\pause
\[
\lim_{x\to 0^+}\sqrt{x}=0=f(0).
\]

\pause
\[
\lim_{x\to 4^-}\sqrt{x}=2=f(4).
\]

\pause
\[
\Rightarrow f \text{ is continuous on } [0,4].
\]
\end{frame}

\section{Rigorous Definition: $\eps$--$\delta$ Continuity}

\begin{frame}{Formal $\eps$--$\delta$ definition}
A function $f$ is continuous at $x=a$ if:

\medskip
For every $\eps>0$, there exists $\delta>0$ such that
\[
|x-a|<\delta \quad \Rightarrow \quad |f(x)-f(a)|<\eps.
\]

\pause
This is exactly the $\eps$--$\delta$ definition of:
\[
\lim_{x\to a} f(x)=f(a).
\]
\end{frame}

\begin{frame}{Example 8: $\eps$--$\delta$ proof (linear function)}
Show that $f(x)=2x+1$ is continuous at $x=3$.

\pause
We must show:
\[
|f(x)-f(3)|<\eps.
\]

\pause
Compute:
\[
|2x+1-7|=2|x-3|.
\]

\pause
To ensure $2|x-3|<\eps$, choose:
\[
\boxed{\delta=\frac{\eps}{2}}.
\]

\pause
Thus $|x-3|<\delta$ implies $|f(x)-f(3)|<\eps$.

\hfill $\square$
\end{frame}

\begin{frame}{Example 9: $\eps$--$\delta$ continuity of $x^2$}
Show that $f(x)=x^2$ is continuous at $x=1$.

\pause
We want:
\[
|x^2-1|<\eps.
\]

\pause
Factor:
\[
|x^2-1|=|x-1||x+1|.
\]

\pause
Assume $|x-1|<1 \Rightarrow x\in(0,2)$ so $|x+1|<3$.

\pause
Thus:
\[
|x^2-1|<3|x-1|.
\]

\pause
Choose:
\[
\boxed{\delta=\min\left\{1,\frac{\eps}{3}\right\}}.
\]

\hfill $\square$
\end{frame}

\section{Summary}

\begin{frame}{Key Takeaways}
\begin{itemize}
  \item Continuity means: limit equals function value.
  \item Three checks: existence of $f(a)$, limit, and equality.
  \item Polynomials and absolute values are continuous everywhere.
  \item Rational functions are continuous where denominators $\neq 0$.
  \item $\eps$--$\delta$ makes continuity mathematically precise.
\end{itemize}
\end{frame}
